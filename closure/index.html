Closure

Closure is a function that has access to the parent scope, even after the parent function has closed.

function outerFunction() {
  const outerVariable = "outerVar";
  function innerFunction() {
    console.log(outerVariable + " called from innerFun");
  }
  innerFunction();
}
outerFunction();

also, its used to keep track of state and to protect private scope

  function counter() {
  let count = 0;
  return function () {
    count++;
    console.log(count);
  };
}
const increment = counter();
increment();
increment();

Game score example

  function createGame(){
    let score = 0;
    function increment(pts){
      score+=pts;
    }
    function decrement(pts){
      score-=pts;
    }
    function getScore(){
      return score;
    }
    return {
      increment,
      decrement,
      getScore
    }
  }
  const game = createGame();
  game.increment(10);
  game.increment(5);
  game.decrement(3);
  console.log(game.getScore());
